import copy


class History:
    listHistory = []
    count = -1

    def __init__(self):
        return

    def addState(self, list):
        self.count += 1
        self.listHistory.append(copy.deepcopy(list))
        return

    def undo(self):
        if self.count == 0:
            return self.listHistory[self.count]
        self.count -= 1
        var = self.listHistory[self.count]
        return var

    def redo(self):
        if self.count == len(self.listHistory) - 1:
            return self.listHistory[self.count]
        self.count += 1
        var = self.listHistory[self.count]
        return var
