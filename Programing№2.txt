def InputNumber(message):
    while True:
        try:
            value = int(input(message))
            break
        except ValueError:
            print("Uncorrect type")
    return value

def MaxElementsInNegativeDiagonal(array):
    currentElement=array[0][0]
    listMinElements=[]
    size=len(array)
    for i in range(size):
        for j in range(size):
            if(i==j):
                currentElement=array[i][j]
                if(currentElement<0):
                    currentMin= array[j][0]
                    for k in range(size):
                        if(currentMin>array[j][k]):
                            currentMin=array[j][k]
                    listMinElements.append(currentMin)

    if(listMinElements==[]):
        print("negative elements are absent")
        return 

    max=listMinElements[0]
    for i in range(len(listMinElements)):      
        if(max<listMinElements[i]):
            max=listMinElements[i]
    print("Maximal elements: ",max)
    return
